# 파이썬 문법
## 입력 받기
```python
# 처음에 숫자, 그 다음에 스트링 문자열 받기
input_list = list()
for _ in range(int(input())):
    input_list.append(input())
```

## List
```python
# list init
a = list() # a = []
a = [0] * 7 # a = [0, 0, 0, 0, 0, 0, 0]

# list append
a = [1, 2, 3]
a.append(4) # a = [1, 2, 3, 4]

# list pop
a = [1, 2, 3]
top = a.pop() # top = 3, a = [1, 2]

# list length
a = [1, 2, 3]
len(a) # 3

# list sort
a = [1, 2, 3]
a.sort() # 오름차순 정렬
a.sort(reverse=True) # 내림차순 정렬
b = sorted(a)
```
